// 1. Tile tap → lazy-load overlay context
const ctx = await avantiActionsApi.getTileContext(tileId);

// 2. Show overlay
// ctx.riskLevel = 'green' | 'yellow' | 'red'

// GREEN → render ctx.summary + ctx.quickPrompts
// No confirm needed. No action card.

// YELLOW → render ctx.preparedAction card
// Show preview, wait for user tap:
await avantiActionsApi.confirm(ctx.preparedAction.actionRunId, {
  confirmed: true
});
await avantiActionsApi.execute(ctx.preparedAction.actionRunId);
// → show Success state

// RED → render blocked explanation
// ctx.preparedAction.status === 'blocked'
// Show safe alternative from ctx.preparedAction.summary
// Never call confirm or execute.

// LOCKED TILE → show upgrade sheet
// tile.isUnlocked === false
// Do not call getTileContext for locked tiles.
